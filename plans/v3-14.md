### 2.2 子项 2 · 死链接扫描

**扫描命令**:

```bash
.venv/Scripts/python.exe -c "
import re
from pathlib import Path

# 扫所有 markdown 里的相对路径链接 [text](path)
md_files = list(Path('.').rglob('*.md'))
md_files = [f for f in md_files if not str(f).startswith('.git')]

dead = []
for f in md_files:
    text = f.read_text(encoding='utf-8', errors='ignore')
    for m in re.finditer(r'\[([^\]]+)\]\(([^)]+)\)', text):
        link = m.group(2)
        if link.startswith(('http://', 'https://', '#', 'mailto:')):
            continue
        # 相对路径
        target = (f.parent / link.split('#')[0]).resolve()
        if not target.exists():
            dead.append((str(f), m.group(0), str(target)))

print(f'死链接 {len(dead)} 处')
for f, raw, target in dead:
    print(f'  {f}: {raw} → {target}')
" > tmp/audit_a2_deadlinks.txt
```
